int maximum(int a[],int n)
{
    int max=a[0],i;
    for(i=1;i<n;i++)
    {
        if(max<a[i])
            max=a[i];
        else
            max=max;
    }
    return max;
}
void manipulation(int a[],int n,int k,int max)
{
    int i,j=1;
    while(j<=k)
    {
        for(i=0;i<n;i++)
        {
            a[i]=max-a[i];
        }
        j++;
        max=maximum(a,n);

    }
    for(i=0;i<n;i++)
    {
        printf("%d ",a[i]);
    }
}
